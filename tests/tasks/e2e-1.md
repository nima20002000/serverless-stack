// Create E2E test Supabase client
export function createE2ESupabaseClient() {
  const url = process.env.E2E_SUPABASE_URL;
  const key = process.env.E2E_SUPABASE_SECRET_KEY;

  if (!url || !key) {
    throw new Error('E2E Supabase credentials not configured');
  }

  // ⚠️ SAFETY CHECK: Prevent production access
  if (url.includes('tanqgnztclrucfldxhuk')) {
    throw new Error('⛔ FATAL: Production Supabase URL detected in E2E tests!');
  }

  return createClient<Database>(url, key);
}

// Get last OTP code for a phone/email (for testing)
export async function getLastOTP(identifier: string): Promise<string> {
  const supabase = createE2ESupabaseClient();

  const { data, error } = await supabase
    .from('otp_codes')
    .select('code')
    .eq('identifier', identifier)
    .order('created_at', { ascending: false })
    .limit(1)
    .single();

  if (error || !data) {
    throw new Error(`Failed to get OTP for ${identifier}: ${error?.message}`);
  }

  return data.code;
}

import type { Page } from '@playwright/test';

/**
 * Mock Zarinpal payment gateway redirect
 * Intercepts redirect to Zarinpal and simulates success callback
 */
export async function mockZarinpalSuccess(page: Page): Promise<void> {
  await page.route('**/zarinpal.com/**', async (route) => {
    const url = new URL(route.request().url());
    const authority = url.searchParams.get('Authority');

    if (!authority) {
      throw new Error('No Authority parameter in Zarinpal redirect');
    }

    console.log(`🔄 Intercepted Zarinpal redirect, Authority: ${authority}`);

    // Simulate success redirect back to site
    const baseURL = process.env.E2E_BASE_URL || 'http://localhost:3000';
    await route.fulfill({
      status: 302,
      headers: {
        Location: `${baseURL}/api/transactions/verify?Authority=${authority}&Status=OK`,
      },
    });
  });
}

/**
 * Mock Digipay payment gateway redirect
 */
export async function mockDigipaySuccess(page: Page): Promise<void> {
  await page.route('**/mydigipay.com/**', async (route) => {
    const url = new URL(route.request().url());
    const trackId = url.searchParams.get('trackId');

    if (!trackId) {
      throw new Error('No trackId in Digipay redirect');
    }

    console.log(`🔄 Intercepted Digipay redirect, trackId: ${trackId}`);

    const baseURL = process.env.E2E_BASE_URL || 'http://localhost:3000';
    await route.fulfill({
      status: 302,
      headers: {
        Location: `${baseURL}/api/transactions/verify-digipay?trackId=${trackId}`,
      },
    });
  });
}

/**
 * Mock payment gateway failure
 */
export async function mockPaymentFailure(
  page: Page,
  gateway: 'zarinpal' | 'digipay'
): Promise<void> {
  const pattern =
    gateway === 'zarinpal' ? '**/zarinpal.com/**' : '**/mydigipay.com/**';

  await page.route(pattern, async (route) => {
    const url = new URL(route.request().url());
    const param =
      gateway === 'zarinpal'
        ? url.searchParams.get('Authority')
        : url.searchParams.get('trackId');

    console.log(`❌ Simulating ${gateway} payment failure`);

    const baseURL = process.env.E2E_BASE_URL || 'http://localhost:3000';
    const endpoint =
      gateway === 'zarinpal'
        ? `/api/transactions/verify?Authority=${param}&Status=NOK`
        : `/api/transactions/verify-digipay?trackId=${param}&status=failed`;

    await route.fulfill({
      status: 302,
      headers: {
        Location: `${baseURL}${endpoint}`,
      },
    });
  });
}
