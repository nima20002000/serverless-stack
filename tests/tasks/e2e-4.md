    await page.fill('[name="firstName"]', 'تست');
    await page.fill('[name="lastName"]', 'تخفیف');
    await page.fill('[name="address"]', 'تهران');
    await page.fill('[name="postalCode"]', '1234567890');

    await page.fill('[name="promoCode"]', 'E2ETEST10');
    await page.click('[data-testid="apply-promo-button"]');

    // Verify promo applied message
    await expect(page.locator('[data-testid="promo-applied"]')).toBeVisible();
    await expect(page.locator('[data-testid="promo-applied"]')).toContainText(
      /10%|۱۰٪/
    );

    await expect(discountAmount).toBeVisible();

    const discountText = await discountAmount.textContent();
    const discount = parseInt(discountText!.replace(/[^\d]/g, ''));

    // Verify discount is 10% of price
    expect(discount).toBe(Math.floor(price * 0.1));

    // Verify total price updated
    const totalElement = page.locator('[data-testid="total-price"]');
    const totalText = await totalElement.textContent();
    const total = parseInt(totalText!.replace(/[^\d]/g, ''));

    await page.click('[data-testid="send-otp-button"]');
    await page.waitForTimeout(1000);
    const { getLastOTP } = await import('../utils/database');
    const otpCode = await getLastOTP(testPhone);
    await page.fill('[name="otp"]', otpCode);
    await page.click('[data-testid="verify-otp-button"]');

    // Complete payment
    await mockZarinpalSuccess(page);
    await page.click('[data-testid="pay-button"]');
    await page.waitForURL(/\/success/);

    // Verify transaction has discount applied
    const transaction = await getLastTransactionByPhone(testPhone);
    expect(transaction.promoCode).toBe('E2ETEST10');
    expect(transaction.discount).toBe(discount);
    expect(transaction.amount).toBe(total);

    // Add 3 different products
    const productCards = page.locator('[data-testid="product-card"]');
    const productCount = await productCards.count();

    const addedProducts = [];
    for (let i = 0; i < Math.min(3, productCount); i++) {
      const card = productCards.nth(i);
      const name = await card
        .locator('[data-testid="product-name"]')
        .textContent();
      const price = await card
        .locator('[data-testid="product-price"]')
        .textContent();

      addedProducts.push({ name, price });

      await card.click();
      await page.click('[data-testid="add-to-cart-button"]');
      await page.goto('/products');
    }

    // Verify cart count
    const cartBadge = page.locator('[data-testid="cart-count"]');
    await expect(cartBadge).toHaveText('3');

    await page.click('[data-testid="cart-icon"]');

    // Verify all products in cart
    const cartItems = page.locator('[data-testid="cart-item"]');
    await expect(cartItems).toHaveCount(3);

    // Verify total price
    let expectedTotal = 0;
    for (const product of addedProducts) {
      const price = parseInt(product.price!.replace(/[^\d]/g, ''));
      expectedTotal += price;

    const totalElement = page.locator('[data-testid="cart-total"]');
    const totalText = await totalElement.textContent();
    const actualTotal = parseInt(totalText!.replace(/[^\d]/g, ''));

    await page.fill('[name="address"]', 'تهران');

    await page.click('[data-testid="send-otp-button"]');
